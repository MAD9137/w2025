# Data Persistence üöÄ

# Introduction

Persistence is the ability to save data across app launches. Swift provides various ways to persist user data, and today we'll explore UserDefaults for simple data storage and SwiftData, Apple's framework for managing structured data.

By the end of this lecture, you‚Äôll:

  1. Understand how to use UserDefaults for lightweight storage.
  2. Learn how to use SwiftData for more structured and complex data persistence.

# 1. UserDefaults: Simple Key-Value Storage

UserDefaults is great for storing lightweight, user-specific data such as settings or preferences.

# How to Use UserDefaults

  1. Saving Data
import SwiftUI

struct UserDefaultsExampleView: View {
    @State private var username: String = ""

    var body: some View {
        VStack(spacing: 20) {
            TextField("Enter your name", text: $username)
                .textFieldStyle(RoundedBorderTextFieldStyle())
                .padding()

            Button("Save Name") {
                UserDefaults.standard.set(username, forKey: "username")
            }

            Button("Retrieve Name") {
                if let savedName = UserDefaults.standard.string(forKey: "username") {
                    username = savedName
                }
            }

            Button("Clear Name") {
                UserDefaults.standard.removeObject(forKey: "username")
            }
        }
        .padding()
    }
}

let isLoggedIn = UserDefaults.standard.bool(forKey: "isLoggedIn")

let levelScores = UserDefaults.standard.array(forKey: "levelScores") as? [Int] ?? []

Singletons in Swift

# What is a Singleton?

A singleton is a design pattern that restricts the instantiation of a class to one object. It ensures that only one instance of the class exists throughout the application's lifecycle. This is useful for objects that need to be shared, such as a network manager or a settings manager, ensuring global access to the instance.

# Creating a Singleton in Swift

To create a singleton, we use a static constant that initializes the instance of the class the first time it‚Äôs accessed.

class NetworkManager {
    // The shared singleton instance
    static let shared = NetworkManager()

    // Private initializer to prevent multiple instances
    private init() {}

    func fetchData() {
        print("Fetching data...")
    }
}

# 2. SwiftData: Structuring Data Without CoreData

SwiftData is designed for managing complex relationships and structured data with ease. It integrates seamlessly with SwiftUI.

# Basic Setup

First, import the framework and define your model.

import SwiftUI
import SwiftData

@Model
class Task {
    var id: UUID
    var title: String
    var isCompleted: Bool

    init(title: String, isCompleted: Bool = false) {
        self.id = UUID()
        self.title = title
        self.isCompleted = isCompleted
    }
}

# Creating and Fetching Data

Here‚Äôs how to work with SwiftData in a SwiftUI view:

import SwiftUI
import SwiftData

struct TasksListView: View {
    @Query var tasks: [Task]
    @Environment(\.modelContext) private var context

    @State private var newTaskTitle: String = ""

    var body: some View {
        NavigationView {
            VStack {
                List {
                    ForEach(tasks) { task in
                        HStack {
                            Text(task.title)
                            Spacer()
                            if task.isCompleted {
                                Image(systemName: "checkmark.circle.fill")
                                    .foregroundColor(.green)
                            }
                        }
                    }
                    .onDelete(perform: deleteTasks)
                }

                HStack {
                    TextField("New Task", text: $newTaskTitle)
                        .textFieldStyle(RoundedBorderTextFieldStyle())
                    Button("Add") {
                        addTask(title: newTaskTitle)
                        newTaskTitle = ""
                    }
                }
                .padding()
            }
            .navigationTitle("Tasks")
        }
    }

    private func addTask(title: String) {
        let newTask = Task(title: title)
        context.insert(newTask)
        try? context.save()
    }

    private func deleteTasks(at offsets: IndexSet) {
        offsets.map { tasks[$0] }.forEach(context.delete)
        try? context.save()
    }
}

# Working with Relationships

Models can reference each other for structured data.

@Model
class Project {
    var name: String
    @Relationship var tasks: [Task] = []

    init(name: String) {
        self.name = name
    }
}

@Model
class Task {
    var title: String
    var isCompleted: Bool
    @Relationship(inverse: \Project.tasks) var project: Project?

    init(title: String, isCompleted: Bool = false) {
        self.title = title
        self.isCompleted = isCompleted
    }
}
