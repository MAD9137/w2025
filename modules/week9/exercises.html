# Exercise 1

You want to create a simple toggle to enable or disable a feature in a SwiftUI view. The state should only be stored in this view. Which property wrapper would you use, and how would you implement it?

Solution

Use @State to manage the local toggle state.

import SwiftUI

struct ToggleFeatureView: View {
    @State private var isFeatureEnabled: Bool = false

    var body: some View {
        VStack {
            Toggle("Enable Feature", isOn: $isFeatureEnabled)
            Text(isFeatureEnabled ? "Feature is ON" : "Feature is OFF")
        }
        .padding()
    }
}

struct ToggleFeatureView_Previews: PreviewProvider {
    static var previews: some View {
        ToggleFeatureView()
    }
}

# Exercise 2

You are building a task list app. Each task is an observable object with a title and completion status. How would you create a TaskItem class and ensure it can be observed for changes?

Solution

Define TaskItem with @Published properties to notify observers when the properties change.

import SwiftUI
import Combine

class TaskItem: ObservableObject, Identifiable {
    let id = UUID()
    @Published var title: String
    @Published var isCompleted: Bool
    
    init(title: String, isCompleted: Bool = false) {
        self.title = title
        self.isCompleted = isCompleted
    }
}

# Exercise 3

You want to display a list of tasks that users can add new items to. The task list should be managed in a single view. Which property wrappers and classes would you use?

Solution

Use an ObservableObject class TaskList to store tasks and a @State property to handle the new task title input.

import SwiftUI

class TaskList: ObservableObject {
    @Published var tasks: [TaskItem] = []
    
    func addTask(title: String) {
        let newTask = TaskItem(title: title)
        tasks.append(newTask)
    }
}

struct TaskListView: View {
    @ObservedObject var taskList = TaskList()
    @State private var newTaskTitle: String = ""

    var body: some View {
        VStack {
            TextField("New Task", text: $newTaskTitle)
                .padding()
                .textFieldStyle(RoundedBorderTextFieldStyle())
            
            Button(action: {
                taskList.addTask(title: newTaskTitle)
                newTaskTitle = ""
            }) {
                Text("Add Task")
            }
            .padding()

            List(taskList.tasks) { task in
                Text(task.title)
            }
        }
        .padding()
    }
}

# Exercise 4

In a settings screen, you want the username and theme color to be accessible across multiple views. How would you implement this using @EnvironmentObject?

Solution

Create an AppSettings class with @Published properties and provide it at the top of your view hierarchy using @EnvironmentObject.

import SwiftUI

class AppSettings: ObservableObject {
    @Published var username: String = "User"
    @Published var themeColor: Color = .blue
}

struct ContentView: View {
    @EnvironmentObject var settings: AppSettings

    var body: some View {
        VStack {
            Text("Hello, \(settings.username)")
                .foregroundColor(settings.themeColor)
            NavigationLink(destination: SettingsView()) {
                Text("Go to Settings")
            }
        }
        .padding()
    }
}

struct SettingsView: View {
    @EnvironmentObject var settings: AppSettings
    @State private var newUsername: String = ""

    var body: some View {
        VStack {
            TextField("New Username", text: $newUsername)
            Button("Update Username") {
                settings.username = newUsername
            }
            ColorPicker("Select Theme Color", selection: $settings.themeColor)
        }
        .padding()
    }
}

@main
struct MyApp: App {
    var body: some Scene {
        WindowGroup {
            ContentView()
                .environmentObject(AppSettings())
        }
    }
}

# Exercise 5

Suppose you want to build an app that displays a stock ticker with a random price update feature. Create a StockTicker class and a TickerView that displays the price and lets users update it.

Solution

Use @Published in StockTicker for price changes and @ObservedObject in TickerView to observe these changes.

import SwiftUI

class StockTicker: ObservableObject {
    @Published var price: Double
    
    init(price: Double) {
        self.price = price
    }
    
    func updatePrice() {
        price = Double.random(in: 100...500)
    }
}

struct TickerView: View {
    @ObservedObject var stockTicker: StockTicker
    
    var body: some View {
        VStack {
            Text("Stock Price: $\(stockTicker.price, specifier: "%.2f")")
                .font(.largeTitle)
            Button("Update Price") {
                stockTicker.updatePrice()
            }
            .padding()
            .background(Color.green)
            .foregroundColor(.white)
            .cornerRadius(8)
        }
        .padding()
    }
}

struct TickerView_Previews: PreviewProvider {
    static var previews: some View {
        TickerView(stockTicker: StockTicker(price: 250))
    }
}

# Exercise 6: Basic Delegation Setup

Define a simple delegate protocol in Swift that has a single method for handling button presses. Implement the protocol in a view that has a button. When the button is pressed, the delegate method should be called.

Solution 
// Define the delegate protocol
protocol ButtonDelegate: AnyObject {
    func didPressButton()
}

// Implement the delegator (the view)
struct ButtonView: View {
    weak var delegate: ButtonDelegate?

    var body: some View {
        Button("Press me") {
            delegate?.didPressButton()
        }
    }
}

// Implement the delegate in another class
struct ContentView: View, ButtonDelegate {
    func didPressButton() {
        print("Button was pressed")
    }

    var body: some View {
        ButtonView(delegate: self)
    }
}

# Exercise 7: Passing Data with a Delegate

Expand on the previous example by adding a text field to ButtonView. The delegate method should now pass the text entered in the text field to the delegate.

Solution 
// Define the delegate protocol
protocol ButtonDelegate: AnyObject {
    func didSubmitText(_ text: String)
}

// Implement the delegator (the view)
struct ButtonView: View {
    weak var delegate: ButtonDelegate?
    @State private var inputText = ""

    var body: some View {
        VStack {
            TextField("Enter text", text: $inputText)
            Button("Submit") {
                delegate?.didSubmitText(inputText)
            }
        }
    }
}

// Implement the delegate in another class
struct ContentView: View, ButtonDelegate {
    func didSubmitText(_ text: String) {
        print("Submitted text: \(text)")
    }

    var body: some View {
        ButtonView(delegate: self)
    }
}

# Exercise 8: Basic Closure Example

Implement a function that takes a closure as a parameter. The closure should be called when a background task finishes (simulated with DispatchQueue.async). Pass a closure that prints "Task complete" when finished.

Solution 
func performTask(completion: @escaping () -> Void) {
    DispatchQueue.global().async {
        // Simulate a background task
        sleep(1)
        DispatchQueue.main.async {
            completion()
        }
    }
}

// Usage
performTask {
    print("Task complete")
}

# Exercise 9: Closure Capturing Values

Create a function that returns a closure, which increments a value by a given amount. Each time the closure is called, the value should be incremented and printed.

Solution 
func makeIncrementer(by amount: Int) -> () -> Void {
    var total = 0
    return {
        total += amount
        print("Total is now \(total)")
    }
}

// Usage
let incrementByFive = makeIncrementer(by: 5)
incrementByFive() // Output: Total is now 5
incrementByFive() // Output: Total is now 10

# Exercise 10: Combining Delegation and Closures

Build a more advanced example where a delegate notifies when a task starts, and a closure is used as a callback when the task finishes. The delegate handles starting the task, while the closure handles completion.

Solution 
// Define the delegate protocol
protocol TaskDelegate: AnyObject {
    func didStartTask()
}

// Implement the task manager
class TaskManager {
    weak var delegate: TaskDelegate?

    func performTask(completion: @escaping () -> Void) {
        delegate?.didStartTask()
        DispatchQueue.global().async {
            // Simulate a background task
            sleep(1)
            DispatchQueue.main.async {
                completion()
            }
        }
    }
}

// Implement the delegate and use the closure
struct ContentView: View, TaskDelegate {
    var taskManager = TaskManager()

    func didStartTask() {
        print("Task started")
    }

    var body: some View {
        Button("Start Task") {
            taskManager.delegate = self
            taskManager.performTask {
                print("Task completed")
            }
        }
    }
}
