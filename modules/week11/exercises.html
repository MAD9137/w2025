# Exercise 1

Display a basic map view centered on a specific location.
Using MapKit in SwiftUI, create a map view centered on New York City with a visible region span of 0.05 for both latitude and longitude.

Solution 
import SwiftUI
import MapKit

struct BasicNYCMapView: View {
    @State private var region = MKCoordinateRegion(
        center: CLLocationCoordinate2D(latitude: 40.7128, longitude: -74.0060),
        span: MKCoordinateSpan(latitudeDelta: 0.05, longitudeDelta: 0.05)
    )

    var body: some View {
        Map(coordinateRegion: $region)
            .edgesIgnoringSafeArea(.all)
    }
}

# Exercise 2

Add annotations to specific locations on the map.
Create a map view with annotations for "Central Park" and "Times Square" in New York City. Display the markers in red.

Solution 
import SwiftUI
import MapKit

struct AnnotatedNYCMapView: View {
    @State private var region = MKCoordinateRegion(
        center: CLLocationCoordinate2D(latitude: 40.7128, longitude: -74.0060),
        span: MKCoordinateSpan(latitudeDelta: 0.1, longitudeDelta: 0.1)
    )
    
    let locations = [
        Location(name: "Central Park", coordinate: CLLocationCoordinate2D(latitude: 40.7851, longitude: -73.9683)),
        Location(name: "Times Square", coordinate: CLLocationCoordinate2D(latitude: 40.7580, longitude: -73.9855))
    ]

    var body: some View {
        Map(coordinateRegion: $region, annotationItems: locations) { location in
            MapMarker(coordinate: location.coordinate, tint: .red)
        }
        .edgesIgnoringSafeArea(.all)
    }
}

# Exercise 3

Create custom annotations for each location.
Modify the annotations to show a custom icon and name for each location. Use a blue star icon and display the location name beneath it.

Solution 
import SwiftUI
import MapKit

struct CustomAnnotatedNYCMapView: View {
    @State private var region = MKCoordinateRegion(
        center: CLLocationCoordinate2D(latitude: 40.7128, longitude: -74.0060),
        span: MKCoordinateSpan(latitudeDelta: 0.1, longitudeDelta: 0.1)
    )
    
    let locations = [
        Location(name: "Central Park", coordinate: CLLocationCoordinate2D(latitude: 40.7851, longitude: -73.9683)),
        Location(name: "Times Square", coordinate: CLLocationCoordinate2D(latitude: 40.7580, longitude: -73.9855))
    ]

    var body: some View {
        Map(coordinateRegion: $region, annotationItems: locations) { location in
            MapAnnotation(coordinate: location.coordinate) {
                VStack {
                    Image(systemName: "star.circle.fill")

                    Text(location.name)
                        .font(.caption)
                        .background(Color.white.opacity(0.7))
                }
            }
        }
        .edgesIgnoringSafeArea(.all)
    }
}

# Exercise 4

Enable user interaction with map annotations.
Create a map view where tapping on a custom annotation displays an alert with the location's name.

Solution 
import SwiftUI
import MapKit

struct InteractiveNYCMapView: View {
    @State private var region = MKCoordinateRegion(
        center: CLLocationCoordinate2D(latitude: 40.7128, longitude: -74.0060),
        span: MKCoordinateSpan(latitudeDelta: 0.1, longitudeDelta: 0.1)
    )
    
    let locations = [
        Location(name: "Central Park", coordinate: CLLocationCoordinate2D(latitude: 40.7851, longitude: -73.9683)),
        Location(name: "Times Square", coordinate: CLLocationCoordinate2D(latitude: 40.7580, longitude: -73.9855))
    ]
    
    @State private var selectedLocation: Location?

    var body: some View {
        Map(coordinateRegion: $region, annotationItems: locations) { location in
            MapAnnotation(coordinate: location.coordinate) {
                Button(action: {
                    selectedLocation = location
                }) {
                    Image(systemName: "star.circle.fill")
                        .foregroundColor(.blue)
                        .font(.largeTitle)
                }
            }
        }
        .edgesIgnoringSafeArea(.all)
        .alert(item: $selectedLocation) { location in
            Alert(title: Text(location.name), message: Text("You selected \(location.name)."), dismissButton: .default(Text("OK")))
        }
    }
}

# Exercise 5

Implement a custom overlay to highlight a specific area.
Create a map view with a circular overlay centered on Central Park with a radius of 1000 meters.

Solution

import MapKit

# Question 6

What is the purpose of CLLocationManager in Core Location, and how do you initialize it in a SwiftUI app?

Solution

The CLLocationManager class is used to request and manage location updates. To initialize it in a SwiftUI app, you create a class conforming to CLLocationManagerDelegate, instantiate CLLocationManager, set its delegate, and request authorization to access location data.

Example:

import SwiftUI
import CoreLocation

class LocationManager: NSObject, ObservableObject, CLLocationManagerDelegate {
    private let locationManager = CLLocationManager()
    
    override init() {
        super.init()
        locationManager.delegate = self
        locationManager.requestWhenInUseAuthorization()
    }
}

# Question 7

How would you set up a SwiftUI view to display the user’s latitude and longitude using LocationManager?

Solution

To display the latitude and longitude in a SwiftUI view, use @StateObject to manage the LocationManager instance. Bind to the location property and show the coordinates in a Text view.

Example:

import SwiftUI

struct ContentView: View {
    @StateObject private var locationManager = LocationManager()
    
    var body: some View {
        VStack {
            if let location = locationManager.location {
                Text("Latitude: \(location.coordinate.latitude, specifier: "%.6f")")
                Text("Longitude: \(location.coordinate.longitude, specifier: "%.6f")")
            } else {
                Text("Locating...")
            }
        }
        .padding()
    }
}

# Question 8

How do you handle errors when location updates fail in CLLocationManagerDelegate? Implement a method to handle location errors with didFailWithError.

Solution

To handle location errors, implement the locationManager(_:didFailWithError:) delegate method. This method provides an Error object detailing what went wrong.

Example:

func locationManager(_ manager: CLLocationManager, didFailWithError error: Error) {
    print("Failed to find user's location: \(error.localizedDescription)")
}

# Question 9

Explain how to use @Published in LocationManager and why it is important for updating the location in SwiftUI.

Solution

The @Published property wrapper allows the location property to automatically notify SwiftUI views of changes. This ensures that whenever location is updated, the SwiftUI view will refresh to show the latest location information.

Example:

@Published var location: CLLocation? = nil

func locationManager(_ manager: CLLocationManager, didUpdateLocations locations: [CLLocation]) {
    location = locations.last
}

# Question 10

Write a function that stops location updates after retrieving a certain number of locations, e.g., 3 locations.

Solution

Create a function that counts the received locations, stopping updates after reaching a specified limit.

Example:

class LocationManager: NSObject, ObservableObject, CLLocationManagerDelegate {
    private let locationManager = CLLocationManager()
    @Published var location: CLLocation? = nil
    private var locationCount = 0
    private let maxLocationUpdates = 3
    
    override init() {
        super.init()
        locationManager.delegate = self
        locationManager.requestWhenInUseAuthorization()
        locationManager.startUpdatingLocation()
    }
    
    func locationManager(_ manager: CLLocationManager, didUpdateLocations locations: [CLLocation]) {
        location = locations.last
        locationCount += 1
        if locationCount >= maxLocationUpdates {
            locationManager.stopUpdatingLocation()
            print("Stopped location updates after \(maxLocationUpdates) updates.")
        }
    }
}
